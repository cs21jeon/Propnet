            const token = localStorage.getItem("proptalk_token");
            fetch("/voiceroom/api/auth/account", {
                method: "DELETE",
                headers: { "Authorization": "Bearer " + token },
            })
            .then(r => r.json())
            .then(data => {
                if (data.ok) {
                    localStorage.removeItem("proptalk_token");
                    localStorage.removeItem("proptalk_refresh_token");
                    localStorage.removeItem("proptalk_user");
                    alert("회원 탈퇴가 완료되었습니다.");
                    window.location.href = "/proptalk/landing";
                } else {
                    alert("탈퇴 실패: " + (data.error || "알 수 없는 오류"));
                }
            })
            .catch(() => alert("회원 탈퇴 중 오류가 발생했습니다."));
        },
